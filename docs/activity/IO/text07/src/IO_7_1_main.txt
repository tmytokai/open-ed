#include "config.h"
#include "setting.h"

void main(void){
    
    init();
    
    __delay_ms(1000); // ATP3011の初期化待ち

    while(1) {
      
        mssp1_i2c_start();
        mssp1_i2c_putch( ( ? << 1) + ? ); // ATP3011のI2Cアドレス(0x2E, 7bit) + write指示(0)
        mssp1_i2c_putch('h'); // 好きな言葉に置き換える
        mssp1_i2c_putch('o');
        mssp1_i2c_putch('g');
        mssp1_i2c_putch('e');
        mssp1_i2c_putch('.');
        mssp1_i2c_putch('\r');
        mssp1_i2c_stop();

        __delay_ms(5000);
    }

   return;
}
