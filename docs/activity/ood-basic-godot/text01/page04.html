ソース 1: キャラクターを右方向に移動距離 100 で動かす 
extends Actor

# export をつけるとTAROのインスペクタで数値を変更できるようになる
@export var speed = 100 

# onready をつけると開始時に表示領域のサイズを取得する
@onready var screen_size = get_viewport_rect().size 

# 各フレームごとに繰り返し実行される関数
func _process(delta):
	# 移動方向・速度をベクトルで指定する
	var velocity = Vector2(1,0)*speed 
	# 位置を更新する
	# deltaは前フレームからの経過時間
	# deltaを速さにかけることでFPSの影響を受けずに移動距離を一定にできる(距離=速さx時間)
	position += velocity*delta
	# 画面外に出ないようにクランプする
	position = position.clamp(Vector2.ZERO, screen_size)
